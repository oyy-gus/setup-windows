function Force-Uninstall($name) {
    $apps = Get-ItemProperty HKLM:\Software\Microsoft\Windows\CurrentVersion\Uninstall\* ,
                             HKLM:\Software\WOW6432Node\Microsoft\Windows\CurrentVersion\Uninstall\* 2>$null |
            Where-Object { $_.DisplayName -like "*$name*" }

    foreach ($app in $apps) {
        if ($app.PSChildName -match "^{.*}$") {
            Write-Host "MSI remove: $($app.DisplayName)"
            Start-Process "msiexec.exe" "/x $($app.PSChildName) /qn /norestart" -Wait
        }
        elseif ($app.UninstallString) {
            Write-Host "Force remove: $($app.DisplayName)"
            $cmd = $app.UninstallString
            if ($cmd -notmatch "/S|/silent|/qn|/quiet") {
                $cmd += " /S /silent /qn /quiet /norestart"
            }
            Start-Process "cmd.exe" "/c $cmd" -Wait
        }
    }
}
